function col2num(col) { # more efficient than an associative array
    if (col == "A") return 1;
    else if (col == "B") return 2;
    else if (col == "C") return 3;
    else if (col == "D") return 4;
    else if (col == "E") return 5;
    else if (col == "F") return 6;
    else if (col == "G") return 7;
    else if (col == "H") return 8;
}
function abs(n) {
    return n > 0 ? n : -n;
}
function sign(n) {
    return n == 0 ? 0 : ((n > 0) * 2 - 1);
}
function isattacking(pcol, prow, tcol, trow) { # piece, target
    # TODO: CLEAN UP GOOD LORD
    if (abs(board[pcol, prow]) == 1) {
        if (sign(board[pcol, prow]) != sign(trow - prow)) return 0;
        # single move, double move, capture
        return (abs(trow - prow) == 1 && !abs(tcol - pcol) && !board[tcol, trow]) ||
        ((prow == 2 || prow == 7) && abs(trow - prow) == 2 && !abs(tcol - pcol) && !board[tcol, trow] && !board[tcol, trow-sign(board[pcol, prow])]) ||
        (abs(trow - prow) == 1 && abs(tcol - pcol) == 1 && board[tcol, trow]);
    } else if (abs(board[pcol, prow]) == 2) {
        if (abs(pcol-tcol) != abs(prow-trow)) return 0;
        while (1) {
            pcol += sign(tcol - pcol);
            prow += sign(trow - prow);
            if (pcol == tcol) return 1;
            else if (board[pcol, prow]) return 0;
        }
    } else if (abs(board[pcol, prow]) == 3) {
        return (abs(tcol - pcol) == 2 && abs(trow - prow) == 1) || (abs(tcol - pcol) == 1 && abs(trow - prow) == 2);
    } else if (abs(board[pcol, prow]) == 4) {
        if (pcol != tcol && prow != trow) return 0;
        while (1) {
            if (prow == trow) pcol += sign(tcol - pcol);
            else prow += sign(trow - prow);
            if (pcol == tcol && prow == trow) return 1;
            else if (board[pcol, prow]) return 0;
        }
    } else if (abs(board[pcol, prow]) == 5) {
        if (abs(pcol-tcol) == abs(prow-trow)) while (1) { # BISHOP MOVEMENT
            pcol += sign(tcol - pcol);
            prow += sign(trow - prow);
            if (pcol == tcol) return 1;
            else if (board[pcol, prow]) return 0;
        } else if (pcol == tcol || prow == trow) while (1) { # ROOK MOVEMENT
            if (prow == trow) pcol += sign(tcol - pcol);
            else prow += sign(trow - prow);
            if (pcol == tcol && prow == trow) return 1;
            else if (board[pcol, prow]) return 0;
        } else return 0;
    } else if (abs(board[pcol, prow]) == 6) {
        return abs(tcol - pcol) <= 1 && abs(trow - prow) <= 1;
    }
    return 0;
}
